   198  func TestString__PreferenceTable(t *testing.T) {
   199  
   200  	cases := [][]string{
   201  		{
   202  			"success",
   203  			"",
   204  			"",
   205  			"'A'\t=>\t'B', 'C', 'D'\n'B'\t=>\t'A', 'C', 'D'\n'C'\t=>\t'A', 'B', 'D'\n'D'\t=>\t'A', 'B', 'C'\n",
   206  		},
   207  		{
   208  			"displays current proposer (middle element)",
   209  			"A",
   210  			"C",
   211  			"'A'\t=>\t'B', 'C'+, 'D'\n'B'\t=>\t'A', 'C', 'D'\n'C'\t=>\t'A', 'B', 'D'\n'D'\t=>\t'A', 'B', 'C'\n",
   212  		},
   213  		{
   214  			"displays current proposer (first element)",
   215  			"A",
   216  			"B",
   217  			"'A'\t=>\t'B'+, 'C', 'D'\n'B'\t=>\t'A', 'C', 'D'\n'C'\t=>\t'A', 'B', 'D'\n'D'\t=>\t'A', 'B', 'C'\n",
   218  		},
   219  		{
   220  			"displays current proposer (last element)",
   221  			"A",
   222  			"D",
   223  			"'A'\t=>\t'B', 'C', 'D'+\n'B'\t=>\t'A', 'C', 'D'\n'C'\t=>\t'A', 'B', 'D'\n'D'\t=>\t'A', 'B', 'C'\n",
   224  		},
   225  	}
   226  
   227  	for i := range cases {
   228  		testCase := cases[i]
   229  
   230  		t.Run(testCase[0], func(t *testing.T) {
   231  			prefs := []MatchPreference{
   232  				{Name: "A", Preferences: []string{"B", "C", "D"}},
   233  				{Name: "B", Preferences: []string{"A", "C", "D"}},
   234  				{Name: "C", Preferences: []string{"A", "B", "D"}},
   235  				{Name: "D", Preferences: []string{"A", "B", "C"}},
   236  			}
   237  
   238  			pt = NewPreferenceTable(&prefs)
   239  
   240  			if len(testCase[1]) > 0 {
   241  				proposed := testCase[1]
   242  				proposer := testCase[2]
   243  				pt[proposed].Accept(pt[proposer])
   244  			}
   245  
   246  			assert.Equal(t, testCase[3], pt.String())
   247  		})
   248  	}
   249  
   250  	t.Run("empty table", func(t *testing.T) {
   251  		prefs := []MatchPreference{}
   252  
   253  		pt = NewPreferenceTable(&prefs)
   254  
   255  		wanted := ""
   256  
   257  		assert.Equal(t, wanted, pt.String())
   258  	})
   259  }
