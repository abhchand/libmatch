   146  func TestReject(t *testing.T) {
   147  	t.Run("success", func(t *testing.T) {
   148  		setupSingleTable()
   149  
   150  		memA.acceptedProposalFrom = &memB
   151  		memB.acceptedProposalFrom = &memA
   152  		memC.acceptedProposalFrom = &memD
   153  		memD.acceptedProposalFrom = &memC
   154  
   155  		memA.Reject(&memC)
   156  
   157  		assert.Equal(t, PreferenceList{members: []*Member{&memB, &memD}}, plA)
   158  		assert.Equal(t, PreferenceList{members: []*Member{&memA, &memC, &memD}}, plB)
   159  		assert.Equal(t, PreferenceList{members: []*Member{&memB, &memD}}, plC)
   160  		assert.Equal(t, PreferenceList{members: []*Member{&memA, &memB, &memC}}, plD)
   161  
   162  		assert.Equal(t, &memB, memA.CurrentProposer())
   163  		assert.Equal(t, &memA, memB.CurrentProposer())
   164  		assert.Equal(t, &memD, memC.CurrentProposer())
   165  		assert.Equal(t, &memC, memD.CurrentProposer())
   166  	})
   167  
   168  	t.Run("Rejecting current proposer", func(t *testing.T) {
   169  		setupSingleTable()
   170  
   171  		memA.acceptedProposalFrom = &memB
   172  		memB.acceptedProposalFrom = &memA
   173  		memC.acceptedProposalFrom = &memD
   174  		memD.acceptedProposalFrom = &memC
   175  
   176  		memA.Reject(&memB)
   177  
   178  		assert.Equal(t, PreferenceList{members: []*Member{&memC, &memD}}, plA)
   179  		assert.Equal(t, PreferenceList{members: []*Member{&memC, &memD}}, plB)
   180  		assert.Equal(t, PreferenceList{members: []*Member{&memA, &memB, &memD}}, plC)
   181  		assert.Equal(t, PreferenceList{members: []*Member{&memA, &memB, &memC}}, plD)
   182  
   183  		/*
   184  		 * A rejects B (its current proposer), so A also has that value reset
   185  		 * B does not lose its current proposer, regardless of who is it
   186  		 */
   187  		assert.Nil(t, memA.CurrentProposer())
   188  		assert.Equal(t, &memA, memB.CurrentProposer())
   189  		assert.Equal(t, &memD, memC.CurrentProposer())
   190  		assert.Equal(t, &memC, memD.CurrentProposer())
   191  	})
   192  }
