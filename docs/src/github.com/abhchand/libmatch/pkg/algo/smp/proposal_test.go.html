   121  func TestSimulateProposal(t *testing.T) {
   122  	t.Run("proposed has no accepted proposal", func(t *testing.T) {
   123  		actualPrefs := []*[]core.MatchPreference{
   124  			{
   125  				{Name: "A", Preferences: []string{"H", "I", "J"}},
   126  				{Name: "B", Preferences: []string{"H", "I", "J"}},
   127  				{Name: "C", Preferences: []string{"H", "I", "J"}},
   128  			},
   129  			{
   130  				{Name: "H", Preferences: []string{"A", "B", "C"}},
   131  				{Name: "I", Preferences: []string{"A", "B", "C"}},
   132  				{Name: "J", Preferences: []string{"A", "B", "C"}},
   133  			},
   134  		}
   135  
   136  		actualTables := core.NewPreferenceTablePair(actualPrefs[0], actualPrefs[1])
   137  
   138  		// C proposes to I, who has no other accepted proposal and will accept
   139  		simulateProposal(actualTables[0]["C"], actualTables[1]["I"])
   140  
   141  		wantedPrefs := []*[]core.MatchPreference{
   142  			{
   143  				{Name: "A", Preferences: []string{"H", "I", "J"}},
   144  				{Name: "B", Preferences: []string{"H", "I", "J"}},
   145  				{Name: "C", Preferences: []string{"H", "I", "J"}},
   146  			},
   147  			{
   148  				{Name: "H", Preferences: []string{"A", "B", "C"}},
   149  				{Name: "I", Preferences: []string{"A", "B", "C"}},
   150  				{Name: "J", Preferences: []string{"A", "B", "C"}},
   151  			},
   152  		}
   153  
   154  		wantedTables := core.NewPreferenceTablePair(wantedPrefs[0], wantedPrefs[1])
   155  
   156  		wantedTables[0]["C"].AcceptMutually(wantedTables[1]["I"])
   157  
   158  		assert.True(t, reflect.DeepEqual(wantedTables, actualTables))
   159  	})

   161  	t.Run("proposed prefers new proposal to existing one", func(t *testing.T) {
   162  		actualPrefs := []*[]core.MatchPreference{
   163  			{
   164  				{Name: "A", Preferences: []string{"H", "I", "J"}},
   165  				{Name: "B", Preferences: []string{"H", "I", "J"}},
   166  				{Name: "C", Preferences: []string{"H", "I", "J"}},
   167  			},
   168  			{
   169  				{Name: "H", Preferences: []string{"A", "B", "C"}},
   170  				{Name: "I", Preferences: []string{"A", "B", "C"}},
   171  				{Name: "J", Preferences: []string{"A", "B", "C"}},
   172  			},
   173  		}
   174  
   175  		actualTables := core.NewPreferenceTablePair(actualPrefs[0], actualPrefs[1])
   176  
   177  		// B proposes to H, then A proposes to H
   178  		// H will prefer the newer proosal (A) and mutually reject the former proposal (B)
   179  		simulateProposal(actualTables[0]["B"], actualTables[1]["H"])
   180  		simulateProposal(actualTables[0]["A"], actualTables[1]["H"])
   181  
   182  		wantedPrefs := []*[]core.MatchPreference{
   183  			{
   184  				{Name: "A", Preferences: []string{"H", "I", "J"}},
   185  				{Name: "B", Preferences: []string{"I", "J"}},
   186  				{Name: "C", Preferences: []string{"H", "I", "J"}},
   187  			},
   188  			{
   189  				{Name: "H", Preferences: []string{"A", "C"}},
   190  				{Name: "I", Preferences: []string{"A", "B", "C"}},
   191  				{Name: "J", Preferences: []string{"A", "B", "C"}},
   192  			},
   193  		}
   194  
   195  		wantedTables := core.NewPreferenceTablePair(wantedPrefs[0], wantedPrefs[1])
   196  
   197  		wantedTables[1]["H"].AcceptMutually(wantedTables[0]["A"])
   198  
   199  		assert.True(t, reflect.DeepEqual(wantedTables[1], actualTables[1]))
