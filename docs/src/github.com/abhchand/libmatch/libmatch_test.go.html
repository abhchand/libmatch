    14  func TestLoad(t *testing.T) {
    15  	t.Run("success", func(t *testing.T) {
    16  		body := `
    17  	  [
    18  	    { "name":"A", "preferences": ["B", "C", "D"] },
    19  	    { "name":"B", "preferences": ["A", "C", "D"] },
    20  	    { "name":"C", "preferences": ["A", "B", "D"] },
    21  	    { "name":"D", "preferences": ["A", "B", "C"] }
    22  	  ]
    23  	  `
    24  
    25  		result, err := Load(strings.NewReader(body))
    26  
    27  		wanted := &[]core.MatchPreference{
    28  			{Name: "A", Preferences: []string{"B", "C", "D"}},
    29  			{Name: "B", Preferences: []string{"A", "C", "D"}},
    30  			{Name: "C", Preferences: []string{"A", "B", "D"}},
    31  			{Name: "D", Preferences: []string{"A", "B", "C"}},
    32  		}
    33  
    34  		assert.Nil(t, err)
    35  		assert.True(t, reflect.DeepEqual(wanted, result))
    36  	})

    39  		// Note missing `:` on final row
    40  		body := `
    41  	  [
    42  	    { "name":"A", "preferences": ["B", "C", "D"] },
    43  	    { "name":"B", "preferences": ["A", "C", "D"] },
    44  	    { "name":"C", "preferences": ["A", "B", "D"] },
    45  	    { "name":"D", "preferences" ["A", "B", "C"] }
    46  	  ]
    47  		`
    48  
    49  		_, err := Load(strings.NewReader(body))
    50  
    51  		assert.Equal(t, "invalid character '[' after object key", err.Error())
    52  	})

    68  	`
